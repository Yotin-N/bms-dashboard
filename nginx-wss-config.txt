# Add this to your Nginx config for WebSocket Secure support

# สำหรับ WebSocket Secure (WSS)
location /bms/socket.io/ {
    proxy_pass https://10.143.106.100:3002/socket.io/;
    proxy_http_version 1.1;
    proxy_set_header Upgrade $http_upgrade;
    proxy_set_header Connection "Upgrade";
    proxy_set_header Host $host;
    proxy_set_header X-Real-IP $remote_addr;
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_set_header X-Forwarded-Proto $scheme;
    proxy_cache off;
    proxy_buffering off;
    proxy_read_timeout 86400s;
    proxy_ssl_verify off; # ถ้า backend ใช้ self-signed cert
}

# หรือถ้า backend ไม่มี SSL ให้ proxy ไป ws:// แล้วให้ Nginx จัดการ SSL
location /bms/socket.io/ {
    proxy_pass http://10.143.106.100:3002/socket.io/;
    proxy_http_version 1.1;
    proxy_set_header Upgrade $http_upgrade;
    proxy_set_header Connection "Upgrade";
    proxy_set_header Host $host;
    proxy_cache off;
    proxy_buffering off;
    proxy_read_timeout 86400s;
}
